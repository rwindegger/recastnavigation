425 inline unsigned int dtIlog2(unsigned int v)
426 {
427  unsigned int r;
428  unsigned int shift;
429  r = (v > 0xffff) << 4; v >>= r;
430  shift = (v > 0xff) << 3; v >>= shift; r |= shift;
431  shift = (v > 0xf) << 2; v >>= shift; r |= shift;
432  shift = (v > 0x3) << 1; v >>= shift; r |= shift;
433  r |= (v >> 1);
434  return r;
435 }
